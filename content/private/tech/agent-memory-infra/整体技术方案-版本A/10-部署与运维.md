## 1. 环境要求

### 1.1 硬件配置（最小生产配置）

| 节点角色 | CPU | 内存 | 磁盘 | 数量 |
|---------|-----|------|------|------|
| K8s Master | 8 核 | 16 GB | 200 GB SSD | 3（HA） |
| App Worker（Agent Framework / API） | 16 核 | 32 GB | 100 GB SSD | 3 |
| Memory Pipeline Worker | 32 核 | 64 GB | 200 GB SSD | 2 |
| PostgreSQL | 16 核 | 64 GB | 1 TB NVMe | 1 主 + 1 从 |
| Neo4j | 32 核 | 128 GB | 2 TB SSD | 1（或 3 节点集群）|
| Redis | 8 核 | 64 GB | 200 GB SSD | 3（Cluster）|
| ClickHouse | 32 核 | 128 GB | 10 TB HDD+SSD | 2 |
| Kafka + ZooKeeper | 8 核 | 32 GB | 500 GB SSD | 3 |

| 组件 | 版本要求 |

| Docker | >= 24.0 |
| Kubernetes | >= 1.28 |
| Helm | >= 3.14 |

## 4. 环境初始化脚本

### 4.1 开发环境一键启动（scripts/init-dev-env.sh）

```bash
#!/usr/bin/env bash
# scripts/init-dev-env.sh
# 一键启动 Agent Memory 开发环境
# 使用方法：bash scripts/init-dev-env.sh

set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_ROOT="$(dirname "$SCRIPT_DIR")"
LOG_FILE="/tmp/agent-memory-init-$(date +%Y%m%d_%H%M%S).log"

# ─── 颜色输出 ─────────────────────────────────────────────────────────────
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
BLUE='\033[0;34m'
NC='\033[0m'

log_info()  { echo -e "${BLUE}[INFO]${NC}  $*" | tee -a "$LOG_FILE"; }
log_ok()    { echo -e "${GREEN}[OK]${NC}    $*" | tee -a "$LOG_FILE"; }
log_warn()  { echo -e "${YELLOW}[WARN]${NC}  $*" | tee -a "$LOG_FILE"; }
log_error() { echo -e "${RED}[ERROR]${NC} $*" | tee -a "$LOG_FILE"; }

# ─── 前置检查 ─────────────────────────────────────────────────────────────
check_prerequisites() {
  log_info "检查前置依赖..."
  local missing=0

  for cmd in docker docker-compose python3 pip; do
    if ! command -v "$cmd" &>/dev/null; then
      log_error "未找到命令: $cmd"
      missing=1
    else
      log_ok "$cmd 已安装: $(${cmd} --version 2>&1 | head -1)"
    fi
  done

  # 检查 Docker daemon 是否运行
  if ! docker info &>/dev/null; then
    log_error "Docker daemon 未运行，请先启动 Docker"
    missing=1
  fi

  # 检查 .env 文件
  if [[ ! -f "${PROJECT_ROOT}/.env" ]]; then
    log_warn ".env 文件不存在，从模板创建..."
    cp "${PROJECT_ROOT}/.env.example" "${PROJECT_ROOT}/.env"

start_infrastructure() {

  cd "$PROJECT_ROOT"

  docker compose up -d \
    zookeeper kafka \
    redis postgres neo4j clickhouse \
    2>&1 | tee -a "$LOG_FILE"

  log_info "等待服务健康检查通过..."
  local services=("am_redis" "am_postgres" "am_neo4j" "am_clickhouse" "am_kafka")
  local max_wait=180
  local waited=0

  for svc in "${services[@]}"; do
    log_info "等待 $svc 健康..."
    while true; do
      status=$(docker inspect --format='{{.State.Health.Status}}' "$svc" 2>/dev/null || echo "unknown")
      if [[ "$status" == "healthy" ]]; then
        log_ok "$svc 健康"
        break
      fi
      if [[ "$waited" -ge "$max_wait" ]]; then
        log_error "$svc 在 ${max_wait}s 内未健康，当前状态: $status"
        docker logs "$svc" --tail 50
        exit 1
      fi
      sleep 5
      waited=$((waited + 5))
    done
    waited=0
  done
}
```

```bash
#!/usr/bin/env bash
# scripts/health-check.sh
# 检查所有组件健康状态
# 使用方法：bash scripts/health-check.sh [--env dev|prod]

set -euo pipefail

ENV="${2:-dev}"
ALL_PASS=true

check_http() {
  local name="$1"
  local url="$2"
  local expected_code="${3:-200}"
  local actual_code
  actual_code=$(curl -s -o /dev/null -w "%{http_code}" --max-time 5 "$url" 2>/dev/null || echo "000")
  if [[ "$actual_code" == "$expected_code" ]]; then
    printf "%-30s %-10s %s\n" "$name" "[OK]" "$url"
  else
    printf "%-30s %-10s %s (HTTP %s)\n" "$name" "[FAIL]" "$url" "$actual_code"
    ALL_PASS=false
  fi
}

check_tcp() {
  local name="$1"
  local host="$2"
  local port="$3"
  if nc -z -w3 "$host" "$port" &>/dev/null; then
    printf "%-30s %-10s %s:%s\n" "$name" "[OK]" "$host" "$port"
  else
    printf "%-30s %-10s %s:%s\n" "$name" "[FAIL]" "$host" "$port"
    ALL_PASS=false
  fi
}

check_redis() {
  local name="$1"
  local url="$2"
  if redis-cli -u "$url" ping 2>/dev/null | grep -q PONG; then
    printf "%-30s %-10s %s\n" "$name" "[OK]" "$url"
  else
    printf "%-30s %-10s %s\n" "$name" "[FAIL]" "$url"
    ALL_PASS=false
  fi
}

echo "====== Agent Memory 健康检查 ======"
echo "环境: $ENV | 时间: $(date)"
echo ""

if [[ "$ENV" == "dev" ]]; then
  REDIS_URL="redis://localhost:6379"
```

## 8. 运维手册

### 8.1 常见故障排查（5 种场景）

#### 故障 1：记忆检索延迟突增（P99 > 1s）

```bash
# Step 1: 定位延迟来源（查看各组件 trace）
curl -s "http://localhost:8000/debug/trace/recent?limit=10" | jq '.[] | {trace_id, total_ms, breakdown}'

# Step 2: 检查 Redis 命中率
redis-cli info stats | grep -E "keyspace_(hits|misses)"

# Step 3: 检查 LanceDB 向量搜索延迟
docker logs am_memory_api 2>&1 | grep "lancedb_search" | tail -20

# Step 4: 检查 Neo4j 慢查询
docker exec am_neo4j cypher-shell -u neo4j -p "$NEO4J_PASS" \
  "CALL dbms.listQueries() YIELD query, elapsedTimeMillis
   WHERE elapsedTimeMillis > 500
   RETURN query, elapsedTimeMillis ORDER BY elapsedTimeMillis DESC LIMIT 10;"

# 常见根因与修复：
# - Redis 内存溢出导致大量驱逐 → 扩容 Redis 或调整 maxmemory-policy
# - LanceDB 索引未构建 → 重建向量索引（make db-rebuild-index）
# - Neo4j 图遍历无索引 → 执行 CREATE INDEX 补建索引
```

#### 故障 2：Memory Pipeline 积压（Kafka lag > 50k）

```bash
# Step 1: 查看 Consumer Group 状态
kafka-consumer-groups.sh --bootstrap-server localhost:9092 \
  --describe --group memory-pipeline-prod

# Step 2: 查看 Pipeline Worker 日志
kubectl logs -n agent-memory -l app=memory-pipeline --tail=100 -f

# Step 3: 检查 LLM 调用是否超时（Extraction/Consolidation 依赖 OpenAI）
kubectl logs -n agent-memory -l app=memory-pipeline --tail=200 \
  | grep -E "(openai|timeout|rate_limit)"

# 修复方案：
# 若因 OpenAI 限流 → 添加指数退避重试，降低并发度
# 若因 Worker 数量不足 → kubectl scale deployment memory-pipeline --replicas=8
# 若因单条 Evidence 处理超时 → 检查并跳过异常记录，写入 DLQ
```

#### 故障 3：PostgreSQL 技能表查询超时

```bash
# Step 1: 查看当前慢查询
psql "$POSTGRES_DSN" -c "
SELECT pid, now() - query_start AS duration, query
FROM pg_stat_activity
WHERE state = 'active' AND now() - query_start > interval '3 seconds'
ORDER BY duration DESC;"

# Step 2: 检查向量索引是否存在
psql "$POSTGRES_DSN" -c "
SELECT indexname, indexdef FROM pg_indexes
WHERE tablename = 'skills' AND indexname LIKE '%embedding%';"

# Step 3: 重建向量索引（如缺失）
psql "$POSTGRES_DSN" -c "
CREATE INDEX CONCURRENTLY idx_skills_embedding
ON skills USING ivfflat (embedding vector_cosine_ops) WITH (lists = 100);"

# Step 4: 检查表膨胀
psql "$POSTGRES_DSN" -c "
SELECT relname, n_dead_tup, n_live_tup,
  round(n_dead_tup * 100.0 / NULLIF(n_live_tup + n_dead_tup, 0), 2) AS dead_pct
FROM pg_stat_user_tables WHERE relname = 'skills';"

# 若 dead_pct > 20%：执行 VACUUM ANALYZE skills;
```

#### 故障 4：Neo4j 内存溢出（OOM Killed）

```bash
# Step 1: 查看 Neo4j 内存使用
kubectl top pods -n agent-memory -l app=neo4j

# Step 2: 检查 Neo4j 内部内存配置
kubectl exec -n agent-memory -l app=neo4j -- \
  cypher-shell -u neo4j -p "$NEO4J_PASS" \
  "CALL dbms.listConfig() YIELD name, value
   WHERE name CONTAINS 'memory' RETURN name, value;"

# Step 3: 临时清理图缓存（不删数据）
kubectl exec -n agent-memory -l app=neo4j -- \
  cypher-shell -u neo4j -p "$NEO4J_PASS" \
  "CALL db.clearQueryCaches();"

# 修复方案：
# - 调大 Neo4j Pod 的 Memory Limit
# - 减小 dbms.memory.heap.max_size 和 dbms.memory.pagecache.size
# - 检查是否有全图扫描的 Cypher 查询（缺少 LIMIT 子句）
```

### 8.3 备份恢复流程

**备份计划**：

| 组件 | 备份频率 | 保留时长 | 备份方式 |
|------|---------|---------|---------|
| PostgreSQL | 每日全量 + 每小时 WAL | 30 天 | pg_dump + WAL 归档到 S3 |
| Neo4j | 每日全量 | 14 天 | neo4j-admin dump |
| LanceDB | 每日快照 | 7 天 | S3 版本控制（自动）|
| Redis | 每 6 小时 RDB | 3 天 | BGSAVE + S3 |

**紧急恢复 RTO/RPO 目标**：

- PostgreSQL：RPO < 1 小时，RTO < 30 分钟
- Neo4j：RPO < 24 小时，RTO < 2 小时
- Redis（Working Memory）：数据丢失可接受（有 TTL），RTO < 5 分钟（重启即可）
